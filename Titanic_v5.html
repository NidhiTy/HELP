high_popln = ['MR','MISS','MRS','MASTER']
def var_surname(df):
    df['Surname'] = df.Name.map(lambda x : re.search("(.*)\\, (.*?)\\.(.*)",str.upper(x)).group(2))
    df['Surname'] = np.where((~df.Surname.isin(high_popln)) & (df.Sex == 'male') ,'MR', \
                              np.where((~df.Surname.isin(high_popln)) & (df.Sex == 'female'),'MRS',df.Surname))
    return df
